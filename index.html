mahasiswa = {
    "Aesa": [85, 90, 88, 92],
    "Agvie": [78, 82, 80, 79],
    "Fitriandhini": [90, 87, 93, 91]
}

def analisis_nilai(data):
    for nama, nilai in data.items():
        rata_rata = sum(nilai) / len(nilai)
        nilai_terkecil = min(nilai)
        nilai_terbesar = max(nilai)

        print("Nama            :", nama)
        print("Nilai           :", nilai)
        print("Rata-rata       :", round(rata_rata, 2))
        print("Nilai terkecil  :", nilai_terkecil)
        print("Nilai terbesar  :", nilai_terbesar)
        print("-" * 30)

if __name__ == "__main__":
    analisis_nilai(mahasiswa)

def input_matriks(baris, kolom, nama):
    matriks = []
    for i in range(baris):
        row = list(map(int, input(f"Baris {i+1}: ").split()))
        matriks.append(row)
    return matriks

def penjumlahan(A, B):
    return [[A[i][j] + B[i][j] for j in range(len(A[0]))] for i in range(len(A))]

def pengurangan(A, B):
    return [[A[i][j] - B[i][j] for j in range(len(A[0]))] for i in range(len(A))]

def perkalian(A, B):
    hasil = []
    for i in range(len(A)):
        row = []
        for j in range(len(B[0])):
            total = 0
            for k in range(len(A[0])):
                total += A[i][k] * B[k][j]
            row.append(total)
        hasil.append(row)
    return hasil

def tampilkan(matriks, judul):
    print("\n", judul)
    for row in matriks:
        print(row)

if __name__ == "__main__":
    baris = int(input("Jumlah baris: "))
    kolom = int(input("Jumlah kolom: "))

    A = input_matriks(baris, kolom, "A")
    B = input_matriks(baris, kolom, "B")

    tampilkan(penjumlahan(A, B), "Penjumlahan")
    tampilkan(pengurangan(A, B), "Pengurangan")

    if len(A[0]) == len(B):
        tampilkan(perkalian(A, B), "Perkalian")
